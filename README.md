## Step 3: Data Adapter Pattern

### Your Data â†’ SentimentReading

Create an adapter that transforms YOUR sentiment data into our format:

```typescript
// adapters/sentimentAdapter.ts
import type { SentimentReading, Regime } from '../types/sentiment';

// Define YOUR existing data shape
interface YourSentimentData {
  timestamp: number;
  sentimentScore: number;      // Your naming
  velocity?: number;
  signalStrength?: number;
  marketState?: string;
  sources?: {
    twitter?: number;
    reddit?: number;
    whaleAlert?: number;
    orderFlow?: number;
  };
  events?: Array<{
    text: string;
    type: string;
    magnitude: number;
  }>;
}

// Adapter function
export function adaptToSentimentReading(data: YourSentimentData): SentimentReading {
  // Map your regime names to ours
  const regimeMap: Record<string, Regime> = {
    'stable': 'calm',
    'normal': 'calm',
    'bullish': 'trending',
    'bearish': 'trending',
    'choppy': 'volatile',
    'high_vol': 'volatile',
    'cascade': 'liquidation',
    'panic': 'liquidation',
  };

  // Calculate attribution percentages
  const sources = data.sources || {};
  const socialRaw = (sources.twitter || 0) + (sources.reddit || 0);
  const onchainRaw = sources.whaleAlert || 0;
  const microRaw = sources.orderFlow || 0;
  const total = socialRaw + onchainRaw + microRaw || 1;

  return {
    timestamp: data.timestamp,
    score: Math.max(-1, Math.min(1, data.sentimentScore)), // Clamp to [-1, 1]
    momentum: data.velocity || 0,
    confidence: data.signalStrength || 0.5,
    regime: regimeMap[data.marketState || 'normal'] || 'calm',
    attribution: {
      social: socialRaw / total,
      onchain: onchainRaw / total,
      microstructure: microRaw / total,
    },
    narrative: data.events?.[0] ? {
      id: `evt-${Date.now()}`,
      summary: data.events[0].text,
      source: mapEventType(data.events[0].type),
      impact: data.events[0].magnitude,
    } : undefined,
  };
}

function mapEventType(type: string): 'social' | 'onchain' | 'microstructure' {
  if (['tweet', 'reddit', 'news', 'influencer'].includes(type)) return 'social';
  if (['whale', 'transfer', 'wallet', 'exchange'].includes(type)) return 'onchain';
  return 'microstructure';
}
```

## Step 4: Connect Your Data Stream

```typescript
// hooks/useSentimentStream.ts
import { useState, useEffect } from 'react';
import type { SentimentReading } from '../types/sentiment';
import { adaptToSentimentReading } from '../adapters/sentimentAdapter';

export function useSentimentStream(wsUrl: string) {
  const [reading, setReading] = useState<SentimentReading | null>(null);
  const [history, setHistory] = useState<SentimentReading[]>([]);
  const [events, setEvents] = useState<SentimentReading['narrative'][]>([]);

  useEffect(() => {
    const ws = new WebSocket(wsUrl);

    ws.onmessage = (event) => {
      const rawData = JSON.parse(event.data);
      const adapted = adaptToSentimentReading(rawData);
      
      setReading(adapted);
      setHistory(prev => [...prev.slice(-199), adapted]);
      
      if (adapted.narrative) {
        setEvents(prev => [adapted.narrative!, ...prev].slice(0, 20));
      }
    };

    return () => ws.close();
  }, [wsUrl]);

  return { reading, history, events };
}
```

```typescript
import { useState, useEffect } from 'react';
import type { SentimentReading } from '../types/sentiment';
import { adaptToSentimentReading } from '../adapters/sentimentAdapter';

export function useSentimentPolling(apiUrl: string, intervalMs = 1000) {
  const [reading, setReading] = useState<SentimentReading | null>(null);
  const [history, setHistory] = useState<SentimentReading[]>([]);

  useEffect(() => {
    const fetchData = async () => {
      const response = await fetch(apiUrl);
      const rawData = await response.json();
      const adapted = adaptToSentimentReading(rawData);
      
      setReading(adapted);
      setHistory(prev => [...prev.slice(-199), adapted]);
    };

    fetchData(); // Initial fetch
    const interval = setInterval(fetchData, intervalMs);
    
    return () => clearInterval(interval);
  }, [apiUrl, intervalMs]);

  return { reading, history };
}
```

## Data Schema Reference

### Complete SentimentReading

```typescript
interface SentimentReading {
  // Required
  timestamp: number;                    // Unix timestamp ms
  score: number;                        // -1.0 to +1.0
  momentum: number;                     // Typically -0.5 to +0.5
  confidence: number;                   // 0.0 to 1.0
  regime: 'calm' | 'trending' | 'volatile' | 'liquidation';
  attribution: {
    social: number;                     // 0.0 to 1.0 (should sum to 1)

  };
  
  // Optional
  narrative?: {
    id: string;                         // Unique identifier
    summary: string;                    // Human-readable description
    source: 'social' | 'onchain' | 'microstructure';
    impact: number;                     // -1.0 to +1.0
    entities?: string[];                // ["Elon Musk", "Binance"]
  };
}
```

### Field Calculations

| Field | How to Calculate |
|-------|------------------|
| `score` | Your raw sentiment normalized to [-1, 1] |
| `momentum` | `(current_score - previous_score) / time_delta` |
| `confidence` | Model confidence, signal-to-noise ratio, or `1 - uncertainty` |
| `regime` | Classify based on volatility thresholds |
| `attribution` | Normalize source weights to sum to 1.0 |
